    void smoothToolpath(ExtrusionPath* path, const PrintConfig& print_config) {

            // Ensure there are enough points to apply smoothing
            if (path->points.size() < 3) {
                return; // Not enough points for smoothing
            }

            // Collect points from the path
            std::vector<tinyspline::rational_t> points;
            for (const Point& p : path->points) {
                points.push_back(static_cast<tinyspline::rational_t>(p.x()));
                points.push_back(static_cast<tinyspline::rational_t>(p.y()));
            }

            // Define the degree of the spline
            std::uint8_t degree = 3; // Cubic spline

            // Fit a B-spline curve
            tinyspline::BSpline spline(degree, 2, points.size() / 2); // 2D points
            spline.set_ctrlp(points);

            // Resample the curve
            std::vector<Point> smoothed_points;
            double u = 0.0;
            double step_size = 0.01; // Sampling rate
            while (u <= 1.0) {
                std::vector<tinyspline::rational_t> result = spline.evaluate(u).result();
                smoothed_points.push_back(Point(static_cast<coord_t>(result[0]), static_cast<coord_t>(result[1])));
                u += step_size;
            }
            // Handle the case where u might not reach exactly 1.0 due to floating-point arithmetic
            if (u > 1.0 && u - step_size < 1.0) {
                std::vector<tinyspline::rational_t> result = spline.evaluate(1.0).result();
                smoothed_points.push_back(Point(static_cast<coord_t>(result[0]), static_cast<coord_t>(result[1])));
            }

            // Replace the original points with the smoothed points
            path->points = smoothed_points;
    }

#include "Octree.hpp"
#include <algorithm>

namespace Slic3r {

    Octree::Octree(const BoundingBox& bounds, int nodeCapacity) : capacity(nodeCapacity) {
        root = new Node(bounds);
    }

    Octree::~Octree() {
        deleteNode(root);
    }

    void Octree::deleteNode(Node* node) {
        if (node) {
            if (!node->isLeaf) {
                for (int i = 0; i < 8; ++i) {
                    deleteNode(node->children[i]);
                }
            }
            delete node;
        }
    }

    void Octree::insert(const Geometry::Point_3& point) {
        insertRecursive(root, point);
    }

    void Octree::insertRecursive(Node* node, const Geometry::Point_3& point) {
        if (!node->bounds.contains(Point(point.x(), point.y())))
            return; // Point is outside the bounds of the node

        if (node->isLeaf) {
            node->points.push_back(point);
            if (node->points.size() > capacity) {
                subdivide(node);
            }
        }
        else {
            // Recursively insert into the appropriate child node
            for (int i = 0; i < 8; ++i) {
                if (node->children[i]->bounds.contains(Point(point.x(), point.y()))) {
                    insertRecursive(node->children[i], point);
                    break;
                }
            }
        }
    }

    void Octree::subdivide(Node* node) {
        // Create eight children
        Vec3f half_size = (node->bounds.max - node->bounds.min) / 2.0f;
        for (int i = 0; i < 8; ++i) {
            Vec3f child_min = node->bounds.min;
            if ((i & 1) != 0) child_min.x() += half_size.x();
            if ((i & 2) != 0) child_min.y() += half_size.y();
            if ((i & 4) != 0) child_min.z() += half_size.z();
            node->children[i] = new Node(BoundingBox(child_min, child_min + half_size));
        }

        // Distribute points to children
        for (const Geometry::Point_3& p : node->points) {
            for (int i = 0; i < 8; ++i) {
                if (node->children[i]->bounds.contains(Point(p.x(), p.y()))) {
                    insertRecursive(node->children[i], p);
                    break;
                }
            }
        }

        node->points.clear();
        node->isLeaf = false;
    }

    void Octree::queryRange(const Geometry::Point_3& center, double radius, std::vector<Geometry::Point_3>& result) const {
        queryRangeRecursive(root, center, radius, result);
    }

    void Octree::queryRangeRecursive(Node* node, const Geometry::Point_3& center, double radius, std::vector<Geometry::Point_3>& result) const {
        // Check if the node's bounding box intersects the query sphere
        if (!node->bounds.intersectsSphere(Point(center.x(), center.y()), center.z(), radius)) {
            return; // No intersection
        }

        if (node->isLeaf) {
            // If it's a leaf node, check each point
            for (const Geometry::Point_3& p : node->points) {
                if (Geometry::distance(Point(center.x(), center.y()), Point(p.x(), p.y())) <= radius) {
                    result.push_back(p);
                }
            }
        }
        else {
            // If it's not a leaf node, recursively query the children
            for (int i = 0; i < 8; ++i) {
                queryRangeRecursive(node->children[i], center, radius, result);
            }
        }
    }

    void Octree::clear() {
        deleteNode(root);
        root = new Node(root->bounds); // Recreate the root node with the same bounds
    }

} // namespace Slic3r

    if (!defined) {
        min = point;
        max = point;
        defined = true;
    } else {
        min.x() = std::min(min.x(), point.x());
        min.y() = std::min(min.y(), point.y());
        max.x() = std::max(max.x(), point.x());
        max.y() = std::max(max.y(), point.y());
    }

    merge(Point(static_cast<coord_t>(point.x()), static_cast<coord_t>(point.y())));

    if (!other.defined)

        merge(other.min);
        merge(other.max);
